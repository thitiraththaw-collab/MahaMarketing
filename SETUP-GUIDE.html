MAHA Marketings - การตั้งค่าระบบ

1. การตั้งค่า Contact Form (Google Sheets)

  1. สร้าง Google Sheet ใหม่
  2. ไปที่ Extensions > Apps Script
  3. คัดลอกโค้ดนี้ลงไปใน `Code.gs`:
const EMAIL_TO = "your-email@gmail.com";

function doPost(e) {
  try {
    const data = JSON.parse(e.postData.contents);
    const sheet = SpreadsheetApp.getActiveSpreadsheet().getActiveSheet();
    const timestamp = new Date();
    
    sheet.appendRow([
      timestamp, data.name, data.company, data.phone, 
      data.line, data.email, data.services, 
      data.total, data.message
    ]);
    
    GmailApp.sendEmail(EMAIL_TO, "MAHA: New Lead from " + data.name, 
      "รายละเอียดลูกค้า:\n" + JSON.stringify(data, null, 2));
    
    return ContentService.createTextOutput(JSON.stringify({"result": "success"}))
      .setMimeType(ContentService.MimeType.JSON);
  } catch (err) {
    return ContentService.createTextOutput(JSON.stringify({"result": "error", "error": err.toString()}))
      .setMimeType(ContentService.MimeType.JSON);
  }
}
  1. กด Deploy > New Deployment เลือกประเภทเป็น Web App
  2. ตั้งค่า "Who has access" เป็น Anyone
  3. คัดลอก Web App URL ไปใส่ใน index.html (หรือผ่าน Admin Panel)

2. วิธี Deploy เว็บไซต์

ใช้ Netlify Drop เพื่อให้เว็บไซต์ออนไลน์ได้ใน 1 นาที:

  1. Export ไฟล์จากหน้า /admin
  2. รวมไฟล์ทั้งหมด (index.html, blog.html, images/) ไว้ในโฟลเดอร์เดียวกัน
  3. ลากโฟลเดอร์ไปวางที่ netlify.com/drop
  4. ได้ URL พร้อมใช้งานทันที!
